              </div>

              <div className="hidden items-center gap-3 lg:flex">
                <div className="flex items-center gap-2 rounded-2xl border border-[#E5D4BA] bg-white/85 p-2 shadow-[0_8px_24px_rgba(0,0,0,0.04)] backdrop-blur-xl">
                  <div className="flex items-center gap-2 px-2 text-xs uppercase tracking-[0.22em] text-[#B98B43]">
                    <Globe2 className="h-4 w-4" />
                    {t.langLabel}
                  </div>
                  <div className="flex items-center gap-1">
                    {(["es", "en", "pt"] as Locale[]).map((lang) => (
                      <button
                        key={lang}
                        type="button"
                        onClick={() => setLocale(lang)}
                        className={`rounded-xl px-3 py-2 text-xs font-semibold uppercase tracking-[0.2em] transition ${
                          locale === lang
                            ? "bg-[linear-gradient(135deg,#C99A5E,#E3C495)] text-[#2B2118]"
                            : "text-[#6F6254] hover:bg-[#FAF2E5] hover:text-[#2B2118]"
                        }`}
                      >
                        {t.langNames[lang]}
                      </button>
                    ))}
                  </div>
                </div>
                <button
                  type="button"
                  onClick={() => navigate("contact")}
                  className="rounded-2xl border border-[#D0A563] bg-[linear-gradient(135deg,#C99A5E,#E3C495)] px-5 py-3 text-sm font-semibold text-[#2B2118] shadow-[0_12px_35px_rgba(201,160,103,0.16)]"
                >
                  {t.nav.contact}
                </button>
              </div>

              <button
                type="button"
                onClick={() => setMobileMenuOpen((prev) => !prev)}
                className="flex h-11 w-11 items-center justify-center rounded-2xl border border-[#E5D4BA] bg-white/88 text-[#5A4938] lg:hidden"
              >
                {mobileMenuOpen ? <X className="h-5 w-5" /> : <Menu className="h-5 w-5" />}
              </button>
            </div>

            {mobileMenuOpen && (
              <div className="mt-4 rounded-[1.75rem] border border-[#E7D6BC] bg-[#FFFDF8]/95 p-4 shadow-[0_16px_40px_rgba(104,73,33,0.08)] lg:hidden">
                <div className="grid gap-2">
                  {pages.map((item) => (
                    <button
                      key={item.key}
                      type="button"
                      onClick={() => navigate(item.key)}
                      className={`rounded-2xl px-4 py-3 text-left text-sm font-medium transition ${
                        page === item.key
                          ? "bg-[#F4E7D2] text-[#2B2118]"
                          : "text-[#6F6254] hover:bg-[#FAF2E5] hover:text-[#2B2118]"
                      }`}
                    >
                      {item.label}
                    </button>
                  ))}
                </div>

                <div className="mt-4 flex items-center justify-between rounded-2xl border border-[#E5D4BA] bg-white/85 p-2">
                  <div className="flex items-center gap-2 px-2 text-xs uppercase tracking-[0.22em] text-[#B98B43]">
                    <Globe2 className="h-4 w-4" />
                    {t.langLabel}
                  </div>
                  <div className="flex items-center gap-1">
                    {(["es", "en", "pt"] as Locale[]).map((lang) => (
                      <button
                        key={lang}
                        type="button"
                        onClick={() => setLocale(lang)}
                        className={`rounded-xl px-3 py-2 text-xs font-semibold uppercase tracking-[0.2em] transition ${
                          locale === lang
                            ? "bg-[linear-gradient(135deg,#C99A5E,#E3C495)] text-[#2B2118]"
                            : "text-[#6F6254] hover:bg-[#FAF2E5] hover:text-[#2B2118]"
                        }`}
                      >
                        {t.langNames[lang]}
                      </button>
                    ))}
                  </div>
                </div>
              </div>
            )}
          </div>
        </header>

        <main>{renderPage()}</main>

        <footer className="border-t border-[#E8D8BF]/70 bg-[#FFFDF8]">
          <div className="mx-auto flex max-w-7xl flex-col gap-6 px-6 py-10 md:flex-row md:items-center md:justify-between">
            <div>
              <p className="text-xs uppercase tracking-[0.42em] text-[#9F7A47]">{t.brand}</p>
              <p className="mt-3 text-sm text-[#7A6A58]">{t.footer.note}</p>
            </div>
            <button
              type="button"
              onClick={() => navigate("home")}
              className="inline-flex items-center gap-2 rounded-2xl border border-[#E5D4BA] bg-white/88 px-5 py-3 text-sm font-semibold text-[#5A4938] shadow-[0_8px_24px_rgba(0,0,0,0.04)] backdrop-blur-xl"
            >
              {t.footer.cta}
              <ArrowRight className="h-4 w-4" />
            </button>
          </div>
        </footer>
      </div>
    </div>
  );
}
